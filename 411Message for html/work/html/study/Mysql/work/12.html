创建tb_worker数据表

create table tb_worker(

id INT(11)PRIMARY KEY AUTO_INCREMENT,

name VARCHAR(30)NOT NUll,

sex ENUM('w','m') DEFAULT 'm',

hobby SET('football','basketball','volleyball'),

score FLOAT(3,1) UNSIGNED,

moblie char(11) UNIQUE NOT NULL,

entry_time DATE

)ENGINE=MyISAM;

查询表结构

desc tb_worker

修改moblie的数据类型

alter table tb_worker MODIFY moblie varchar(11);

在moblie后添加字段

alter table tb_worker add intro text after moblie;
