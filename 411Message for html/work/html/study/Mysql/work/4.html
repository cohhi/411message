创建数据库:create database students;

创建表

CREATE TABLE `students`.`student` (
`studentno` INT UNSIGNED COMMENT '学号',
`loginpwd` VARCHAR (20) DEFAULT '00000' COMMENT '密码',
`name` VARCHAR (20) COMMENT '姓名',
`sex` TINYINT (1) COMMENT '性别',
`gradeid` INT,
`phone` VARCHAR (20) NOT NULL COMMENT '手机号码',
`address` VARCHAR (255) NOT NULL COMMENT '住址',
`borndate` DATETIME COMMENT '出生日期',
`email` VARCHAR (50) NOT NULL COMMENT '邮箱',
`identitieard` VARCHAR (18) COMMENT '身份证号码'
) ENGINE = INNODB CHARSET = utf8;

添加数据后通过 select * from student 查看表
