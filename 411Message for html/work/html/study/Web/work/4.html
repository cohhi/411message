浮动的属性 float
float:left; //向左浮动
float:right; //向右浮动
float:none; //不浮动

清除浮动 clear;
clear:right;
clear:left;
clear:both;

溢出overflow
overflow:visible; //内容不会被修剪,会呈现在元素狂之外
overflow:hidden; //内容被修剪,不会显示外部内容
overflow:scroll; //内容被修剪,会显示滚动条
overflow:auto; //如果过长,如果需要,显示滚动条

position
position:static //默认属性
position:relative //相对自己的位置
position:absolute //相对父类定位
position:fixed //相对浏览器窗口定位
